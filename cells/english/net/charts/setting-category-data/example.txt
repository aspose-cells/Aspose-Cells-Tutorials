//ExStepSummary:1: This step instantiates a new Workbook object.
//ExStepSummary:2: This step obtains a reference to the first worksheet in the workbook.
//ExStepSummary:3: This step adds sample numeric values to specific cells in the worksheet.
//ExStepSummary:4: This step adds category labels to the worksheet cells.
//ExStepSummary:5: This step creates a chart in the worksheet and configures it with a specific data range.
//ExStepSummary:6: This step sets the category data for the chart's series collection.
//ExStepSummary:7: This step saves the workbook as an Excel file with the specified name.

//ExStepImage:3:images/2.png

//ExStep:0-
// Specify the output directory for the Excel file
string outputDir = "Your Document Directory";

//ExStep:1-
// Instantiate a new Workbook object
Workbook workbook = new Workbook();

//ExStep:2-
// Get a reference to the first worksheet
Worksheet worksheet = workbook.Worksheets[0];

//ExStep:3-
// Add sample numeric data to cells
worksheet.Cells["A1"].PutValue(10);
worksheet.Cells["A2"].PutValue(100);
worksheet.Cells["A3"].PutValue(170);
worksheet.Cells["A4"].PutValue(200);
worksheet.Cells["B1"].PutValue(120);
worksheet.Cells["B2"].PutValue(320);
worksheet.Cells["B3"].PutValue(50);
worksheet.Cells["B4"].PutValue(40);

//ExStep:4-
// Add category labels to the worksheet
worksheet.Cells["C1"].PutValue("Q1");
worksheet.Cells["C2"].PutValue("Q2");
worksheet.Cells["C3"].PutValue("Y1");
worksheet.Cells["C4"].PutValue("Y2");

//ExStep:5-
// Create a column chart
int chartIndex = worksheet.Charts.Add(Aspose.Cells.Charts.ChartType.Column, 5, 0, 15, 5);

//ExStep:6-
// Access the newly created chart
Aspose.Cells.Charts.Chart chart = worksheet.Charts[chartIndex];

//ExStep:7-
// Add data source for the chart
chart.NSeries.Add("A1:B4", true);
// Set category data for the chart
chart.NSeries.CategoryData = "C1:C4";

//ExStep:8-
// Save the Excel file
workbook.Save(outputDir + "outputSettingCategoryData.xlsx");

//ExStep:9-
// Print success message
Console.WriteLine("SettingCategoryData executed successfully.");
//ExEnd
