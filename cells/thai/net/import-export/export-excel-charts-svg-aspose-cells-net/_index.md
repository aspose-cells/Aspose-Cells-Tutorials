## คู่มือการใช้งาน
### การส่งออกแผนภูมิ Excel เป็น SVG
เป้าหมายหลักคือการส่งออกแผนภูมิจากเวิร์กบุ๊ก Excel ไปยังไฟล์ SVG โดยใช้ Aspose.Cells นี่คือวิธีดำเนินการดังกล่าว:

#### 1. โหลดเวิร์กบุ๊กและเข้าถึงเวิร์กชีต
เริ่มต้นด้วยการโหลดไฟล์ Excel ของคุณลงใน `Workbook` วัตถุและเข้าถึงแผ่นงานที่ต้องการที่มีแผนภูมิ
```csharp
string sourceDir = RunExamples.Get_SourceDirectory();
string outputDir = RunExamples.Get_OutputDirectory();

// สร้างสมุดงานจากไฟล์ Excel ที่มีอยู่
Workbook workbook = new Workbook(sourceDir + "sampleExportChartToSvgWithViewBox.xlsx");

// เข้าถึงแผ่นงานแรก
Worksheet worksheet = workbook.Worksheets[0];
```
#### 2. การเข้าถึงและกำหนดค่าตัวเลือกการส่งออกแผนภูมิ
ระบุแผนภูมิที่คุณต้องการส่งออก จากนั้นกำหนดค่าโดยใช้ `ImageOrPrintOptions`-
```csharp
Aspose.Cells.Charts.Chart chart = worksheet.Charts[0];

// ตั้งค่าตัวเลือกภาพหรือการพิมพ์โดยเปิดใช้งาน SVGFitToViewPort
Aspose.Cells.Rendering.ImageOrPrintOptions opts = new Aspose.Cells.Rendering.ImageOrPrintOptions();
opts.SaveFormat = SaveFormat.Svg;
opts.SVGFitToViewPort = true; // ช่วยให้แน่ใจว่าแผนภูมิพอดีกับช่องมองภาพ
```
#### 3. ส่งออกแผนภูมิไปยัง SVG
สุดท้าย ให้บันทึกแผนภูมิเป็นไฟล์ SVG
```csharp
// บันทึกแผนภูมิในรูปแบบ SVG
cart.ToImage(outputDir + "outputExportChartToSvgWithViewBox.svg", opts);

Console.WriteLine("ExportChartToSvgWithViewBox executed successfully.");
```
