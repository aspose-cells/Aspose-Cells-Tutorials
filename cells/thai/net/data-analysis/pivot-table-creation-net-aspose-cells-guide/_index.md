## คู่มือการใช้งาน
### การสร้าง PivotTable ด้วย Aspose.Cells
เรามาดูขั้นตอนการสร้าง PivotTable ทีละขั้นตอนโดยใช้ Aspose.Cells สำหรับ .NET กัน

#### ขั้นตอนที่ 1: เริ่มต้นสมุดงานของคุณ
ขั้นแรก ให้สร้างอินสแตนซ์ของ `Workbook` คลาส นี่แสดงถึงไฟล์ Excel ของคุณ:

```csharp
// การสร้างอินสแตนซ์ของวัตถุเวิร์กบุ๊ก
Workbook workbook = new Workbook();
```

#### ขั้นตอนที่ 2: เตรียมข้อมูลในเวิร์กชีต
เข้าถึงเวิร์กชีตแรกและป้อนข้อมูลที่จำเป็นสำหรับ PivotTable ของคุณ:

```csharp
// การรับข้อมูลอ้างอิงของแผ่นงานที่เพิ่มใหม่
Worksheet sheet = workbook.Worksheets[0];
Cells cells = sheet.Cells;

// การกำหนดค่าให้กับเซลล์
cells["A1"].PutValue("Sport");
cells["B1"].PutValue("Quarter");
cells["C1"].PutValue("Sales");

// การเพิ่มข้อมูลตัวอย่าง
string[] sports = { "Golf", "Golf", "Tennis", "Tennis", "Tennis", "Tennis", "Golf" };
string[] quarters = { "Qtr3", "Qtr4", "Qtr3", "Qtr4", "Qtr3", "Qtr4", "Qtr3" };
int[] sales = { 1500, 2000, 600, 1500, 4070, 5000, 6430 };

for (int i = 0; i < sports.Length; i++)
{
    cells[$"A{i + 2}"].PutValue(sports[i]);
cells[$"B{i + 2}"].PutValue(quarters[i]);
cells[$"C{i + 2}"].PutValue(sales[i]);
}
```

#### ขั้นตอนที่ 3: สร้างและกำหนดค่า PivotTable
ตอนนี้ เพิ่ม PivotTable ลงในเวิร์กชีตของคุณ:

```csharp
// การเพิ่ม PivotTable ลงในเวิร์กชีต
PivotTableCollection pivotTables = sheet.PivotTables;
int index = pivotTables.Add("=A1:C8", "E3", "PivotTable2");

// การเข้าถึงอินสแตนซ์ของ PivotTable ที่เพิ่มเข้ามาใหม่
PivotTable pivotTable = pivotTables[index];

// การกำหนดค่าการตั้งค่า PivotTable
pivotTable.RowGrand = false; // ซ่อนยอดรวมทั้งหมดสำหรับแถว

// การลากฟิลด์ไปยังพื้นที่ที่เหมาะสม
pivotTable.AddFieldToArea(PivotFieldType.Row, 0);   // สนามกีฬาในพื้นที่แถวบ้าน
pivotTable.AddFieldToArea(PivotFieldType.Column, 1); // ฟิลด์ไตรมาสในพื้นที่คอลัมน์
pivotTable.AddFieldToArea(PivotFieldType.Data, 2);   // ฟิลด์การขายในพื้นที่ข้อมูล
```

#### ขั้นตอนที่ 4: บันทึกสมุดงาน
สุดท้ายให้บันทึกสมุดงานของคุณเพื่อดูผลลัพธ์:

```csharp
// การบันทึกไฟล์ Excel
cells.Workbook.Save("pivotTable_test_out.xls");
```
