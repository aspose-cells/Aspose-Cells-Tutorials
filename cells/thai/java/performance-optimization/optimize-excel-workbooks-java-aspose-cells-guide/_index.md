# เพิ่มประสิทธิภาพเวิร์กบุ๊ก Excel ใน Java โดยใช้ Aspose.Cells: คู่มือประสิทธิภาพ

## การแนะนำ
กำลังดิ้นรนเพื่อจัดการเวิร์กบุ๊ก Excel ขนาดใหญ่อย่างมีประสิทธิภาพภายในแอปพลิเคชัน Java ของคุณอยู่ใช่หรือไม่ บทช่วยสอนที่ครอบคลุมนี้จะสาธิตวิธีใช้ **Aspose.Cells สำหรับ Java** เพื่อเพิ่มประสิทธิภาพการจัดการสมุดงานของคุณ โดยใช้ประโยชน์จากการปรับแต่ง `LightCellsDataProvider`เราจะสำรวจเทคนิคในการปรับปรุงการทำงาน ลดการใช้หน่วยความจำ และเพิ่มประสิทธิภาพการทำงาน

คู่มือนี้ถือว่าคุณมีความรู้พื้นฐานเกี่ยวกับ Java และคุ้นเคยกับ Maven หรือ Gradle มาเริ่มกันเลย!

## คู่มือการใช้งาน
เราจะแบ่งการใช้งานออกเป็นคุณสมบัติที่แตกต่างกันเพื่อความชัดเจนและง่ายต่อการเข้าใจ

### คุณลักษณะที่ 1: สร้างตัวอย่างและกำหนดค่าสมุดงาน
#### ภาพรวม
ฟีเจอร์นี้สาธิตการสร้างอินสแตนซ์ใหม่ของ Aspose.Cells `Workbook` และกำหนดค่าการนับแผ่นงาน
```java
import com.aspose.cells.Workbook;
// สร้างเวิร์กบุ๊กใหม่ด้วยเวิร์กชีตหนึ่งแผ่นตามค่าเริ่มต้น
Workbook wb = new Workbook();
int sheetCount = 1; // ปรับเปลี่ยนตามความจำเป็น
```
#### ตัวเลือกการกำหนดค่า
- ปรับเปลี่ยน `sheetCount` เพื่อให้มีจำนวนแผ่นงานตามที่ต้องการในเบื้องต้น

### คุณลักษณะที่ 2: เพิ่มและกำหนดค่าแผ่นงาน
#### ภาพรวม
ที่นี่ เราเพิ่มเวิร์กชีตใหม่ลงในเวิร์กบุ๊ก ตั้งชื่อ และกำหนดความกว้างของคอลัมน์เพื่อจัดระเบียบข้อมูลได้ดีขึ้น
```java
import com.aspose.cells.Worksheet;
import com.aspose.cells.Cells;
for (int k = 0; k < sheetCount; k++) {
    Worksheet sheet = null;
    if (k == 0) {
        // เปลี่ยนชื่อเวิร์กชีตแรกเป็น "ทดสอบ"
        sheet = wb.getWorksheets().get(k);
        sheet.setName("test");
    } else {
        // เพิ่มเวิร์กชีตใหม่และตั้งชื่อให้เหมาะสม
        int sheetIndex = wb.getWorksheets().add();
        sheet = wb.getWorksheets().get(sheetIndex);
        sheet.setName("test" + sheetIndex);
    }
    
    Cells cells = sheet.getCells();
    // กำหนดความกว้างของคอลัมน์สำหรับ 15 คอลัมน์แรกเป็น 15 หน่วย
    for (int j = 0; j < 15; j++) {
        cells.setColumnWidth(j, 15);
    }
}
```
#### ตัวเลือกการกำหนดค่าคีย์
- ปรับ `sheet.getName()` เพื่อให้เหมาะกับรูปแบบการตั้งชื่อของคุณ
- แก้ไข `cells.setColumnWidth()` ตามความต้องการในการนำเสนอข้อมูล

### คุณสมบัติที่ 3: รวมเซลล์และเพิ่มไฮเปอร์ลิงก์
#### ภาพรวม
ในส่วนนี้จะแสดงตัวอย่างการผสานเซลล์ในรูปแบบเฉพาะ และการเพิ่มไฮเปอร์ลิงก์ภายในและภายนอก
```java
import com.aspose.cells.HyperlinkCollection;
int rowCount = 100000; // กำหนดจำนวนแถวสำหรับการดำเนินการ
for (int k = 0; k < sheetCount; k++) {
    Worksheet sheet = wb.getWorksheets().get(k);
    Cells cells = sheet.getCells();
    HyperlinkCollection hyperlinks = sheet.getHyperlinks();

    // รวมและเพิ่มไฮเปอร์ลิงก์ใน 10 คอลัมน์แรก
    for (int i = 0; i < rowCount; i++) {
        for (int j = 0; j < 10; j++) {
            if (j % 3 == 0) {
                cells.merge(i, j, 1, 2);
            }
            
            if (i % 50 == 0) {
                if (j == 0) {
                    hyperlinks.add(i, j, 1, 1, "test!A1");
                } else if (j == 3) {
                    hyperlinks.add(i, j, 1, 1, "http://www.google.com");
                }
            }
        }
    }

    // รวมเซลล์ในชุดคอลัมน์ที่สอง
    for (int i = 0; i < rowCount; i++) {
        for (int j = 10; j < 20; j++) {
            if (j == 12) {
                cells.merge(i, j, 1, 3);
            }
        }
    }
}
```
#### ข้อควรพิจารณาที่สำคัญ
- ใช้ `cells.merge()` เพื่อจัดกลุ่มข้อมูลอย่างมีตรรกะภายในเวิร์กบุ๊กของคุณ
- ใช้ประโยชน์ `hyperlinks.add()` เพื่อเชื่อมโยงข้อมูลที่เกี่ยวข้องข้ามแผ่นงานหรือทรัพยากรภายนอก

### คุณสมบัติที่ 4: กำหนดค่าและบันทึกสมุดงานด้วยผู้ให้บริการข้อมูล LightCells
#### ภาพรวม
ฟีเจอร์สุดท้ายนี้สาธิตการตั้งค่าแบบกำหนดเอง `LightCellsDataProvider` เพื่อบันทึกสมุดงานขนาดใหญ่ได้อย่างมีประสิทธิภาพ โดยลดการใช้หน่วยความจำลงอย่างมาก
```java
import com.aspose.cells.OoxmlSaveOptions;
import com.example.LightCellsDataProviderDemo; // แทนที่ด้วยเส้นทางนำเข้าจริงสำหรับคลาสผู้ให้บริการข้อมูลของคุณ

LightCellsDataProviderDemo dataProvider = new LightCellsDataProviderDemo(wb, 1, rowCount, 20);
OoxmlSaveOptions opt = new OoxmlSaveOptions();
opt.setLightCellsDataProvider(dataProvider);

String outDir = "YOUR_OUTPUT_DIRECTORY";
wb.save(outDir + "/Demo_out.xlsx", opt);
```
#### ตัวเลือกการกำหนดค่าคีย์
- ปรับแต่ง `LightCellsDataProviderDemo` เพื่อจัดการกับข้อมูลเฉพาะอย่างมีประสิทธิภาพ
- ใช้ `OoxmlSaveOptions.setLightCellsDataProvider()` เพื่อการประหยัดที่เหมาะสมที่สุด
