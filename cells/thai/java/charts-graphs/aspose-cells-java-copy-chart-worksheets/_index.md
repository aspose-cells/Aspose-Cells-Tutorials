# เรียนรู้การใช้ Aspose.Cells ใน Java: คัดลอกแผนภูมิระหว่างเวิร์กชีต

## การแนะนำ

พร้อมที่จะปรับปรุงการจัดการแผนภูมิของคุณหรือยัง มาเริ่มต้นด้วยข้อกำหนดเบื้องต้นกันเลย!

## คู่มือการใช้งาน

มาใช้งานฟังก์ชันการคัดลอกแผนภูมิจากเวิร์กชีตหนึ่งไปยังอีกเวิร์กชีตหนึ่งกัน

### การคัดลอกแผนภูมิระหว่างแผ่นงาน

#### ภาพรวม:
คุณลักษณะนี้ช่วยให้คุณสามารถทำซ้ำแผนภูมิบนเวิร์กชีตต่างๆ ในเวิร์กบุ๊ก Excel ได้ ช่วยให้จัดการการนำเสนอข้อมูลได้อย่างมีประสิทธิภาพ

##### ขั้นตอนที่ 1: โหลดเวิร์กบุ๊ก
เริ่มต้นด้วยการโหลดเวิร์กบุ๊กที่มีอยู่ของคุณซึ่งมีแผนภูมิ
```java
import com.aspose.cells.Workbook;
import AsposeCellsExamples.Utils;

public class CopyChartExample {
    public static void main(String[] args) throws Exception {
        // กำหนดเส้นทางไปยังไดเร็กทอรีเอกสาร
        String dataDir = Utils.getSharedDataDir(CopyChartExample.class) + "TechnicalArticles/";

        // โหลดสมุดงานที่มีอยู่จากเส้นทางที่ระบุ
        Workbook workbook = new Workbook(dataDir + "Shapes.xls");
    }
}
```

##### ขั้นตอนที่ 2: เข้าถึงแผ่นงานและแผนภูมิ
เข้าถึงแผ่นงานแหล่งที่มาที่มีแผนภูมิของคุณ
```java
import com.aspose.cells.WorksheetCollection;
import com.aspose.cells.Worksheet;

// รับชุดเอกสารประกอบการสอนจากสมุดงาน
WorksheetCollection ws = workbook.getWorksheets();

// ดึงข้อมูลเวิร์กชีตที่มาชื่อ "แผนภูมิ"
Worksheet sheet1 = ws.get("Chart");
```

##### ขั้นตอนที่ 3: คัดลอกแผนภูมิไปยังเวิร์กชีตอื่น
ใช้ `addCopy` วิธีการจำลองแผนภูมิบนเวิร์กชีตอื่น
```java
import com.aspose.cells.Chart;
import com.aspose.cells.ChartShape;

// รับแผนภูมิแรกจากแผ่นงานต้นฉบับ
Chart chart = sheet1.getCharts().get(0);

// เข้าถึงรูปร่างแผนภูมิ
ChartShape cshape = chart.getChartObject();

// แผ่นงานเป้าหมายที่จะคัดลอกแผนภูมิลงใน
Worksheet sheet2 = ws.get("Result");

// คัดลอกแผนภูมิไปยังเวิร์กชีตเป้าหมายในตำแหน่งที่ระบุ
sheet2.getShapes().addCopy(cshape, 20, 0, 2, 0);
```

##### ขั้นตอนที่ 4: บันทึกสมุดงาน
สุดท้าย ให้บันทึกสมุดงานของคุณด้วยแผนภูมิที่คัดลอกใหม่
```java
// กำหนดเส้นทางเอาต์พุตและบันทึกสมุดงานที่แก้ไข
temporaryFile = dataDir + "CCFOneWToAnother_out.xls";
workbook.save(temporaryFile);
```
